package com.yh.jdbc.queryome;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.Statement;

public class QueryOne {
	//查询数据库表中的数据总条数
	public static void main(String[] args) {
		String url = "jdbc:mysql://localhost:3306/login";
		String user = "root";
		String password = "123456";

		// 1.加载驱动，把com.mysql.jdbc.Driver这分字节码加载进JVM，当字节码加载到JVM后，就会执行该字节码的静态代码块
		// java1.6之后，不加这句话也行。在javaweb中必须加
		Class.forName("com.mysql.jdbc.Driver");
		// 2.获取连接对象
		Connection conn = DriverManager.getConnection(url, user, password);
		// 3.编写sql语句
		String sql = "SELECT count(*) from enp_tab";
		// 把sql放在数据库里执行，并返回执行结果 statement表示连接数据库，由数据库执行指令
		Statement st = conn.createStatement();
		// 4.执行sql st对象调用方法
		ResultSet res = st.executeQuery(sql);
		if(res.next()) {
			int count = res.getInt(1);
			System.out.println(count);
		}
		st.close();
		conn.close();

	}
	}

}
